% --------------------------- Slide 2 — Problem & Motivation ---------------------------
\begin{frame}{Problem \& Motivation}
  \centering
  \begin{tabularx}{0.92\textwidth}{@{}p{2.4cm} L@{}}
    \toprule
    \hp{Task}        & \textbf{Binary classification} — predict whether a Home Credit applicant defaults  (\texttt{TARGET} $\in \{0, 1\}$). \\[0.4em]
    \hp{Data}        & 307{,}511 applicants $\times$ 122 features  (Kaggle, real-world). \\[0.4em]
    \hp{Challenge}   & Severe class imbalance — only \hi{8.07\%} default; a ``predict-no-default'' baseline scores 92\% accuracy with 0\% recall. \\[0.4em]
    \hp{Approach}    & Compare classic ML (LR, RF, HGBC) and neural networks (Shallow / Deep MLP) under the same pipeline; then study imbalance-handling on the best NN. \\[0.4em]
    \hp{Metric}      & ROC AUC  (accuracy is misleading on imbalanced data). \\
    \bottomrule
  \end{tabularx}

  \bigskip
  \begin{center}
    \itshape Research question: do classic shallow learners and neural networks\\[0.1em]
    respond differently to class imbalance handling?
  \end{center}
\end{frame}

% --------------------------- Slide 4 — Classic ML ---------------------------
\begin{frame}{Classic ML Models}
  \centering
  \small
  \begin{tabularx}{0.95\textwidth}{@{}l L r@{}}
    \toprule
    \hp{Model (family)}                                  & \hp{Best hyperparameters}                                                              & \hp{Val AUC} \\
    \midrule
    Logistic Regression \emph{(linear)}                  & \texttt{C}=0.1; \texttt{class\_weight=balanced}                                        & 0.7487 \\[0.3em]
    Random Forest \emph{(bagging)}                       & \texttt{n\_estimators}=200; \texttt{min\_samples\_split}=100; \texttt{class\_weight=balanced} & 0.7470 \\[0.3em]
    HistGradientBoosting \emph{(boosting)}               & \texttt{learning\_rate}=0.05; \texttt{max\_iter}=200; \texttt{min\_samples\_leaf}=100   & \hi{\textbf{0.7595}} \\
    \bottomrule
  \end{tabularx}

  \bigskip
  \normalsize
  \raggedright
  \hp{Key takeaways}
  \begin{itemize}
    \item \texttt{class\_weight='balanced'} on LR / RF — handles imbalance directly in the loss.
    \item HGBC wins by $\sim$1 AUC point — boosted trees usually dominate tabular data.
    \item RF overfits at \texttt{leaf=1}: train AUC 0.96 vs val 0.747; regularized leaf wins.
  \end{itemize}
\end{frame}

  \bigskip
  \hp{Key takeaways}
  \begin{itemize}

  \end{itemize}

% --------------------------- Slide 7 — Imbalance Handling ---------------------------
\begin{frame}{Imbalance Handling on the Best MLP}
  \centering
  \small
  \begin{tabularx}{\textwidth}{@{}l C C C C C C@{}}
    \toprule
    \hp{Method}              & \hp{Threshold} & \hp{ROC AUC} & \hp{PR AUC} & \hp{Precision} & \hp{Recall} & \hp{F1} \\
    \midrule
    Baseline                 & 0.500 & 0.745 & 0.222 & 0.00 & 0.00 & 0.00 \\
    SMOTE                    & 0.500 & {\color{warn}\textbf{0.648}} & 0.134 & 0.16 & 0.21 & 0.18 \\
    Threshold tuning $\star$ & 0.164 & 0.745 & 0.222 & 0.24 & \hi{\textbf{0.40}} & \textbf{0.30} \\
    \bottomrule
  \end{tabularx}

  \bigskip
  \normalsize
  \raggedright
  \hp{Key takeaways}
  \begin{itemize}
    \item Baseline at 0.5 cutoff: AUC 0.745 but recall \hi{0\%}  (no prediction crosses 0.5).
    \item SMOTE hurts: AUC drops 0.10.
    \item Threshold tuning at 0.164 lifts recall to \hi{40\%}.
  \end{itemize}

  \medskip
  \begin{center}{\color{accent}\itshape Choosing the threshold matters more than rebalancing the data.}\end{center}
\end{frame}

% --------------------------- Slide 8 — Conclusions ---------------------------
\begin{frame}{Conclusions}
  \hp{What we learned}
  \begin{itemize}
    \item HistGradientBoosting wins overall  (val AUC $=$ \hi{0.7595}).
    \item Neural networks tie LR / RF — depth alone does not help on tabular data.
    \item Threshold tuning lifts MLP recall from \hi{0\% to 40\%}; SMOTE hurt AUC.
  \end{itemize}

  \bigskip
  \hp{Limitations}
  \begin{itemize}
    \item Only main \texttt{application\_*.csv} tables  (auxiliary tables typically add several AUC points).
    \item \texttt{PredefinedSplit} is fast but optimistic; full $k$-fold CV would tighten CIs.
  \end{itemize}
\end{frame}
