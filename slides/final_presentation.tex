% --------------------------- Slide 2 — Problem ---------------------------
\begin{frame}{Problem \& Motivation}
  \textbf{The task} \quad Predict whether a loan applicant defaults  (\texttt{TARGET} $\in \{0, 1\}$).

  \bigskip
  \begin{tabular}{@{}p{0.30\textwidth}@{\quad}p{0.62\textwidth}@{}}
    \toprule
    \hp{Default rate}      & only \hi{8.07\%} of applicants  (24{,}825 / 307{,}511) \\
    \hp{Naive baseline}    & ``always predict no default'' $\to$ 92\% accuracy, \hi{0\% recall} \\
    \hp{Primary metric}    & \hi{ROC AUC}  (accuracy is misleading on imbalanced data) \\
    \bottomrule
  \end{tabular}

  \bigskip
  \begin{center}
    {\color{accent}\itshape Do classic shallow learners and neural networks\\[0.1em]
    respond differently to class imbalance handling?}
  \end{center}
\end{frame}

% --------------------------- Slide 4 — Pipeline ---------------------------
\begin{frame}{Shared Pipeline}
  \begin{tabular}{@{}p{0.22\textwidth}@{\quad}p{0.70\textwidth}@{}}
    \toprule
    \hp{Step}       & \hp{Operation} \\
    \midrule

% --------------------------- Slide 5 — Classic ML ---------------------------
\begin{frame}{Classic ML Models}
  \small
  \begin{tabular}{@{}lll r@{}}
    \toprule
    \hp{Model}                & \hp{Family}        & \hp{Tuned hyperparameters}                & \hp{Val AUC} \\
    \midrule
    Logistic Regression       & Linear             & \texttt{C}, \texttt{tol},  \texttt{class\_weight=balanced}            & 0.7487 \\
    Random Forest             & Tree (bagging)     & \texttt{n\_estimators}, \texttt{min\_samples\_*}                       & 0.7470 \\
    HistGradientBoosting      & Tree (boosting)    & \texttt{learning\_rate}, \texttt{max\_iter}, \texttt{min\_samples\_leaf} & \hi{\textbf{0.7595}} \\
    \bottomrule
  \end{tabular}

  \bigskip
  \normalsize
  \hp{Findings}
  \begin{itemize}
    \item \hi{Boosting wins} — typical for tabular data.
    \item RF overfits at \texttt{leaf=1}  (train 0.96 vs val 0.747); the regularized config wins.
    \item LR is competitive — most signal is linear in \texttt{EXT\_SOURCE\_*}.
  \end{itemize}
\end{frame}

% --------------------------- Slide 6 — Neural Network ---------------------------
\begin{frame}{Neural Network Models}
  \small
  \begin{tabular}{@{}lll r@{}}
    \toprule
    \hp{Architecture}              & \hp{Hidden}        & \hp{Parameters}   & \hp{Val AUC} \\
    \midrule
    Shallow MLP                    & $(64,)$            & $\sim$15.7K       & 0.7405 \\
    Deep MLP                       & $(128, 64, 32)$    & $\sim$42K         & \hi{\textbf{0.7416}} \\
    \bottomrule
  \end{tabular}

  \bigskip
  \normalsize
  Both: ReLU \,·\, Adam \,·\, early stopping. Tuned over
  \texttt{alpha} $\in \{10^{-4}, 10^{-3}\}$, \texttt{lr} $\in \{0.001, 0.005\}$.
  Best for both: $\alpha = 10^{-3}$, \texttt{lr} $= 0.005$.

  \bigskip
  \hp{Findings}
  \begin{itemize}
    \item \hi{Depth doesn't help}: $0.7405$ vs $0.7416$  (within noise).
    \item Tripling parameters bought \emph{nothing}.
    \item Both NNs land below the classic learners.
  \end{itemize}
\end{frame}

% --------------------------- Slide 8 — Imbalance ---------------------------
\begin{frame}{Imbalance Handling on the Best MLP}
  \small
  \begin{tabular}{@{}lcccccc@{}}
    \toprule
    \hp{Method}              & \hp{Threshold} & \hp{ROC AUC} & \hp{PR AUC} & \hp{Precision} & \hp{Recall} & \hp{F1} \\
    \midrule
    Baseline                 & 0.500 & 0.745 & 0.222 & 0.00 & 0.00 & 0.00 \\
    SMOTE                    & 0.500 & {\color{warn}\textbf{0.648}} & 0.134 & 0.16 & 0.21 & 0.18 \\
    Threshold tuning $\star$ & \hi{0.164} & 0.745 & 0.222 & 0.24 & \hi{\textbf{0.40}} & \hi{\textbf{0.30}} \\
    \bottomrule
  \end{tabular}

  \bigskip
  \normalsize
  \hp{Findings}
  \begin{itemize}
    \item Baseline at 0.5 cutoff: AUC 0.745 but \hi{recall 0\%}  (no prediction crosses 0.5).
    \item SMOTE \hi{hurts}: AUC drops 0.10.
    \item Threshold tuning at 0.164 lifts recall to \hi{40\%}.
  \end{itemize}

  \medskip
  \begin{center}{\color{accent}\itshape Choosing the threshold matters more than rebalancing the data.}\end{center}
\end{frame}

% --------------------------- Slide 9 — Conclusions ---------------------------
\begin{frame}{Conclusions}
  \hp{What we learned}
  \begin{itemize}
    \item \hi{HistGradientBoosting} wins  (val AUC $= 0.7595$).
    \item \hi{Neural networks} tie LR / RF — depth alone does not help on tabular data.
    \item \hi{Threshold tuning} lifts MLP recall from 0\% to 40\%; \hi{SMOTE} hurt AUC.
  \end{itemize}

  \bigskip
  \hp{Limitations}
  \begin{itemize}
    \item Only main \texttt{application\_*.csv} tables  (auxiliary tables typically add several AUC points).
    \item \texttt{PredefinedSplit} is fast but optimistic; full $k$-fold CV would tighten CIs.
  \end{itemize}

  \bigskip
  \hp{Future work}
  \begin{itemize}
    \item Auxiliary tables  \,·\,  ensemble HGBC $+$ best MLP  \,·\,  Platt / isotonic calibration.
  \end{itemize}
\end{frame}
