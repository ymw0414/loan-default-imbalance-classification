% ---------- Slide 2 — Business Problem and Motivation (Nathaniel) ----------
\begin{frame}{Business Problem and Motivation}

\textbf{Goal}

Predict the probability that a loan applicant will default before making a lending decision.

\vspace{0.4cm}

\textbf{Why it matters}
\begin{itemize}
    \item False negatives: approve borrowers who default
    \item False positives: reject creditworthy borrowers
    \item Models help balance risk vs. profitability
\end{itemize}

\vspace{0.4cm}

\begin{block}{Key challenge}
Default events are rare (\textasciitilde 8\%) $\Rightarrow$ strong class imbalance
\end{block}

\end{frame}


% ---------- Slide 3 — From Imbalance to Research Question (Nathaniel) ----------
\begin{frame}{From Imbalance to Research Question}

\scriptsize

\textbf{Why imbalance matters}
\begin{itemize}
    \item Models can achieve high ROC AUC but still predict no defaults
    \item Default threshold (0.5) often fails in imbalanced settings
\end{itemize}

\textbf{Common solutions}
\begin{itemize}
    \item Reweighting the loss (class weights)
    \item Resampling the data (e.g., SMOTE)
    \item Adjusting the decision threshold
\end{itemize}

\begin{block}{Research Question}
Do neural networks respond to class imbalance handling techniques the same way as classic shallow learners?
\end{block}

\textbf{Models compared}
\begin{itemize}
    \item Logistic Regression, Random Forest, HistGradientBoosting
    \item Shallow and Deep MLP
\end{itemize}

\end{frame}


% ---------- Slide 4 — Dataset Overview (Nathaniel) ----------
\begin{frame}{Dataset Overview}

\textbf{What the dataset represents}
\begin{itemize}
    \item Loan applications from Home Credit

\end{itemize}

\vspace{0.2cm}

\textbf{Main feature groups}

\end{frame}

% ---------- Slide 6 — Classic ML Strategy (Nathaniel) ----------
\begin{frame}{Classic ML Strategy}

\scriptsize

\textbf{Why start with classic ML?}
\begin{itemize}
    \item The data is tabular: rows are applicants, columns are features
    \item Classical models are strong baselines for tabular prediction
    \item They provide a benchmark for the neural networks
\end{itemize}

\textbf{Models tested}
\begin{itemize}
    \item \textbf{Logistic Regression}: strong baseline for probability prediction
    \item \textbf{Random Forest}: captures nonlinear patterns and reduces overfitting
    \item \textbf{Histogram Gradient Boosting}: captures complex interactions efficiently
\end{itemize}

\begin{block}{Handling class imbalance}
\begin{itemize}
    \item Logistic Regression and Random Forest give more weight to default cases (\texttt{class\_weight='balanced'}).
    \item Histogram Gradient Boosting used without class weights (boosting still captures rare cases well).
\end{itemize}
\end{block}

\end{frame}

\end{itemize}

% --------------------------- Slide 11 — Conclusions ---------------------------
\begin{frame}{Conclusions}
  \scriptsize
  \hp{Best model overall}
  \begin{itemize}
    \item \textbf{HGBC wins}  (val AUC $=$ \hi{0.7595}) — nonlinear tree boosting captures feature interactions.
    \item \textbf{LR \& RF}  close behind — LR benefits from strong linear predictors (\texttt{EXT\_SOURCE\_*}).
    \item \textbf{Deep MLP} (0.7416) $\approx$ Shallow MLP (0.7405) — depth doesn't help.
    \item Both NNs lag classic shallow learners by $\sim$1--2 AUC pts.
  \end{itemize}

  \medskip
  \hp{Imbalance handling on the MLP}
  \begin{itemize}
    \item Threshold tuning lifts recall \hi{0\% $\to$ 41\%}; SMOTE hurts AUC.
    \item Choosing the threshold matters more than rebalancing the data.
  \end{itemize}

  \medskip
  \hp{Limitations}
  \begin{itemize}
    \item Only main \texttt{application\_*.csv} tables  (auxiliary tables would add several AUC points).
    \item Single train/val split, not full $k$-fold CV.
  \end{itemize}
\end{frame}
