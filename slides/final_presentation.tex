% ---------- Slide 2 — Business Problem and Motivation (Nathaniel) ----------
\begin{frame}{Business Problem and Motivation}

\textbf{Goal}

Predict the probability that a loan applicant will default before making a lending decision.

\vspace{0.4cm}

\textbf{Why it matters}
\begin{itemize}
    \item False negatives: approve borrowers who default
    \item False positives: reject creditworthy borrowers
    \item Models help balance risk vs. profitability
\end{itemize}

\vspace{0.4cm}

\begin{block}{Key challenge}
Default events are rare (\textasciitilde 8\%) $\Rightarrow$ strong class imbalance
\end{block}

\end{frame}


% ---------- Slide 3 — From Imbalance to Research Question (Nathaniel) ----------
\begin{frame}{From Imbalance to Research Question}

\scriptsize

\textbf{Why imbalance matters}
\begin{itemize}
    \item Models can achieve high ROC AUC but still predict no defaults
    \item Default threshold (0.5) often fails in imbalanced settings
\end{itemize}

\textbf{Common solutions}
\begin{itemize}
    \item Reweighting the loss (class weights)
    \item Resampling the data (e.g., SMOTE)
    \item Adjusting the decision threshold
\end{itemize}

\begin{block}{Research Question}
\begin{itemize}
    \item Do neural networks respond to class imbalance handling techniques the same way as classic shallow learners?
\end{itemize}
\end{block}

\textbf{Models compared}
\begin{itemize}
    \item Logistic Regression, Random Forest, HistGradientBoosting
    \item Shallow and Deep MLP
\end{itemize}

\end{frame}

% ---------- Slide 6 — Classic ML Strategy (Nathaniel) ----------
\begin{frame}{Classic ML Strategy}

\scriptsize

\textbf{Why start with classic ML?}
\begin{itemize}
    \item The data is tabular: rows are applicants, columns are features
    \item Classical models are strong baselines for tabular prediction
    \item They provide a benchmark for the neural networks
\end{itemize}

\vspace{0.2cm}

\textbf{Models tested}
\begin{itemize}
    \item \textbf{Logistic Regression}: strong baseline for probability prediction
    \item \textbf{Random Forest}: captures nonlinear patterns and reduces overfitting
    \item \textbf{Histogram Gradient Boosting}: captures complex interactions efficiently
\end{itemize}

\vspace{0.2cm}

\begin{block}{Handling class imbalance}
\begin{itemize}
    \item Logistic Regression and Random Forest give more weight to default cases (\texttt{class\_weight='balanced'}).
    \item Histogram Gradient Boosting used without class weights (boosting still captures rare cases well).
\end{itemize}
\end{block}

\end{frame}

\textbf{Key takeaways}
\begin{itemize}
    \item Histogram Gradient Boosting performs best (0.7595), suggesting boosted trees capture nonlinear interactions in tabular data effectively.

    \item Logistic Regression and Random Forest perform similarly, indicating strong linear signal alongside moderate nonlinearities.

    \item Random Forest shows signs of overfitting (train 0.96 vs val 0.747), while boosting remains more stable.
\end{itemize}

% ---------- Slide 9 — Imbalance Handling (Minwoo, leakage-fixed) ----------
\begin{frame}{Imbalance Handling on the Best MLP}

\scriptsize

\textbf{Three strategies compared} (sklearn \texttt{MLPClassifier} does not support \texttt{class\_weight})

\begin{tabular}{@{}lcccccc@{}}
    \toprule
    \hp{Method}              & \hp{Threshold} & \hp{ROC AUC} & \hp{PR AUC} & \hp{Precision} & \hp{Recall} & \hp{F1} \\
    \midrule
    Baseline                 & 0.500 & 0.742 & 0.217 & 0.00 & 0.00 & 0.00 \\
    SMOTE                    & 0.500 & {\color{warn}\textbf{0.648}} & 0.134 & 0.16 & 0.21 & 0.18 \\
    Threshold tuning $\star$ & 0.139 & 0.742 & 0.217 & 0.22 & \hi{\textbf{0.41}} & \textbf{0.29} \\
    \bottomrule
\end{tabular}

\vspace{0.2cm}

\textbf{Key takeaways}
\begin{itemize}
    \item Baseline at 0.5: AUC 0.742 but recall \hi{0\%}
    \item SMOTE hurts: AUC drops 0.09
    \item Threshold tuning at 0.139 lifts recall to \hi{41\%}
\end{itemize}

\vspace{0.1cm}

\begin{block}{Headline}
Choosing the threshold matters more than rebalancing the data.
\end{block}

\end{frame}

% ---------- Slide 11 — Why NN Underperforms (Minwoo) ----------
\begin{frame}{Why NN Underperforms on Tabular Data}

\scriptsize

\textbf{1. Tabular favors trees, not NN}
\begin{itemize}
    \item NN excels on image, text, audio — continuous, structured inputs
    \item Trees split on individual features cleanly; NN must weight all at once
\end{itemize}

\textbf{2. Feature signal is already linear-friendly}
\begin{itemize}
    \item \texttt{EXT\_SOURCE\_*} are pre-engineered credit scores — LR / trees use them directly
    \item NN must rediscover the linear relationship through nonlinear layers
\end{itemize}

\textbf{3. One-hot encoding penalizes NN}
\begin{itemize}
    \item 16 categorical columns $\to$ 140 sparse 0/1 columns (70\% of features)
    \item Trees handle sparse binary inputs natively; MLP must learn weights on all
\end{itemize}

\textbf{4. sklearn MLP has no \texttt{class\_weight}}
\begin{itemize}
    \item LR / RF apply \texttt{class\_weight='balanced'} in the loss; MLP cannot
\end{itemize}

\begin{block}{Consensus in the literature}
Tree-based models still outperform deep learning on tabular data
(Grinsztajn et al., NeurIPS 2022).
\end{block}

\end{frame}


% ---------- Slide 12 — Conclusions ----------
\begin{frame}{Conclusions}

\footnotesize

\textbf{What we found}
\begin{itemize}
    \item HGBC wins (val AUC $=$ \hi{0.7595}) — tree boosting captures feature interactions
    \item Neural networks don't help on this tabular data — both MLPs lag classic shallow learners by $\sim$1--2 AUC pts (and even depth doesn't rescue them: Deep MLP $\approx$ Shallow MLP, within noise)
\end{itemize}

\end{frame}
