% --------------------------- Slide 2 — Problem ---------------------------
\begin{frame}{Problem \& Motivation}
  \textbf{\color{gwnavy}\large The task}\\
  Predict the probability that a loan applicant defaults
  (binary classification, \texttt{TARGET}~$\in \{0, 1\}$).

  \bigskip
  \textbf{\color{gwnavy}\large Why this matters}
  \begin{itemize}
    \item Lenders need accurate risk scoring to balance approval volume and credit losses.
    \item False negatives (approving a bad loan) are far more costly than false positives.
  \end{itemize}

  \bigskip
  \textbf{\color{gwnavy}\large Class imbalance challenge}
  \begin{itemize}
    \item Only $\sim$\textbf{8.07\%} of applicants default — the data is severely imbalanced.
    \item A naive ``always predict no default'' baseline scores 92\% accuracy with \textbf{zero recall}.
    \item Evaluating with accuracy alone is misleading; we use \textbf{ROC AUC} as the primary metric.
  \end{itemize}

  \bigskip
  \textbf{\color{gwnavy}\large Research question}\\
  {\color{gwbuff}\itshape\large Do classic shallow learners and neural networks
  respond differently to class imbalance handling?}
\end{frame}

      \textbf{\color{gwnavy}Class imbalance}

      \textbf{\color{gwnavy}Strongest signals}
      \begin{itemize}\small
        \item \texttt{EXT\_SOURCE\_1/2/3} external credit scores ($|\rho| \approx 0.16$--$0.18$)
        \item \texttt{DAYS\_EMPLOYED}\,$=\,$365{,}243 sentinel flags 18\% (unemployed)
        \item Building-info columns are 50--70\% missing
      \end{itemize}

% --------------------------- Slide 4 — Shared Pipeline ---------------------------
\begin{frame}{Shared Pipeline}
  \textbf{\color{gwnavy}\large Preprocessing (identical for all five models)}
  \begin{itemize}
    \item \textbf{80/20 stratified} train--val split (\texttt{random\_state = 42})
    \item Drop \texttt{SK\_ID\_CURR}; keep IDs only for the submission file
    \item Mean imputation for numeric NaN; one-hot encode 16 categoricals $\to$ \textbf{244 features}
    \item \texttt{StandardScaler} fitted on train only (essential for LR / MLP)
  \end{itemize}

  \bigskip
  \textbf{\color{gwnavy}\large Tuning \& validation}
  \begin{itemize}
    \item \texttt{GridSearchCV} with \texttt{PredefinedSplit} (single train$\to$val fold, no random k-fold)
    \item Scoring: \texttt{roc\_auc}; best parameter setting picked per model
    \item CV CSVs saved to \texttt{result/home\_credit/cv\_results/GridSearchCV/}
  \end{itemize}
\end{frame}

    \textbf{HistGradientBoosting}    & \texttt{learning\_rate} $\in \{0.01, 0.05, 0.1\}$, \texttt{max\_iter} $\in \{100, 200\}$, \texttt{min\_samples\_leaf} $\in \{20, 100\}$ & \textbf{0.7595} \\

  \vspace{1.0em}
  \normalsize
  \textbf{\color{gwnavy}Findings}
  \begin{itemize}\small
    \item \textbf{HistGradientBoosting} is the strongest classic learner — boosted trees usually dominate tabular credit data.
    \item \textbf{Logistic Regression} (linear) with \texttt{class\_weight=balanced} is competitive — most signal is linear in $\texttt{EXT\_SOURCE\_*}$.
    \item \textbf{Random Forest} overfits aggressively (train AUC 0.96 vs val 0.747) when \texttt{min\_samples\_leaf=1}; the regularized config wins.
  \end{itemize}
\end{frame}

    \textbf{Deep MLP} -- hidden $(128, 64, 32)$      & same grid as above & \textbf{0.7416} \\
    \bottomrule
  \end{tabularx}

  \vspace{1.0em}
  \normalsize
  \textbf{\color{gwnavy}Findings}
  \begin{itemize}\small
    \item \textbf{Best for both:} $\alpha = 10^{-3}$, \texttt{lr\_init} $= 0.005$ — stronger L2 regularization wins.
    \item \textbf{Depth doesn't help:} $0.7405$ vs $0.7416$ — a 0.001 difference, well within noise; tripling parameters bought nothing.
    \item Both NNs land slightly \emph{below} the classic shallow learners — the bottleneck is the feature set, not the model class.
  \end{itemize}
\end{frame}

% --------------------------- Slide 9 — Conclusions ---------------------------
\begin{frame}{Conclusions}
  \textbf{\color{gwnavy}\large What we learned}
  \begin{itemize}\small
    \item \textbf{HistGradientBoosting} wins overall at val ROC AUC $= 0.7595$ — tree boosting takes the lead by $\sim$1 AUC point.
    \item \textbf{Neural networks} tie LR and RF — depth alone does not unlock new performance on this tabular data.
    \item {\color{gwbuff}\textbf{Threshold tuning} on the MLP lifts recall from 0\% to 40\%; SMOTE hurt AUC.}
  \end{itemize}

  \bigskip
  \textbf{\color{gwnavy}\large Limitations}
  \begin{itemize}\small
    \item Only main \texttt{application\_*.csv} tables — auxiliary tables typically add several AUC points.
    \item \texttt{PredefinedSplit} is fast but optimistic; full $k$-fold CV would tighten confidence intervals.
  \end{itemize}

  \bigskip
  \textbf{\color{gwnavy}\large Future work}
  \begin{itemize}\small
    \item Integrate auxiliary tables \,·\, ensemble HGBC $+$ best MLP \,·\, Platt / isotonic calibration.
  \end{itemize}
\end{frame}
