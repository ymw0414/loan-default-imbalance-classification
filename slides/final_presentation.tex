% --------------------------- Slide 2 — Problem & Motivation ---------------------------
\begin{frame}{Problem \& Motivation}
  \begin{description}[Challenge]
    \item[\hp{Task}]      Binary classification — predict loan default  (\texttt{TARGET} $\in \{0, 1\}$)
    \item[\hp{Data}]      307{,}511 applicants $\times$ 122 features  (Kaggle)
    \item[\hp{Challenge}] only \hi{8.07\%} default $\Rightarrow$ severe class imbalance
    \item[\hp{Approach}]  5 models  (3 classic ML + 2 NN)  +  imbalance handling on best NN
    \item[\hp{Metric}]    ROC AUC  (accuracy is misleading on imbalanced data)
  \end{description}

  \medskip
  \begin{center}
    {\color{accent}\itshape Research question: do classic ML and NN respond differently to imbalance handling?}
  \end{center}
\end{frame}

% --------------------------- Slide 7 — Imbalance ---------------------------
\begin{frame}{Imbalance Handling on the Best MLP}
  \centering
  \footnotesize
  \begin{tabular}{@{}lcccccc@{}}
    \toprule
    \hp{Method}              & \hp{Thresh.} & \hp{AUC} & \hp{PR-AUC} & \hp{Prec.} & \hp{Recall} & \hp{F1} \\
    \midrule
    Baseline                 & 0.500 & 0.742 & 0.217 & 0.00 & 0.00 & 0.00 \\
    SMOTE                    & 0.500 & {\color{warn}\textbf{0.648}} & 0.134 & 0.16 & 0.21 & 0.18 \\
    Threshold tuning $\star$ & 0.139 & 0.742 & 0.217 & 0.22 & \hi{\textbf{0.41}} & \textbf{0.29} \\
    \bottomrule
  \end{tabular}

  \medskip
  \footnotesize
  \raggedright
  \hp{Key takeaways}
  \begin{itemize}
    \item Baseline at 0.5: AUC 0.742 but recall \hi{0\%}.
    \item SMOTE hurts: AUC drops 0.09.
    \item Threshold tuning at 0.139 lifts recall to \hi{41\%}.
  \end{itemize}

  \medskip
  \small
  \begin{center}{\color{accent}\itshape Threshold $>$ resampling.}\end{center}
\end{frame}

% --------------------------- Slide 8 — Conclusions ---------------------------
\begin{frame}{Conclusions}
  \footnotesize
  \hp{Best model overall}\\
  HistGradientBoosting takes the top spot at val ROC AUC $=$ \hi{0.7595},
  indicating that nonlinear tree boosting captures interactions across
  heterogeneous features. LR and RF perform similarly, both slightly below
  HGBC — LR likely benefits from strong linear predictors such as
  \texttt{EXT\_SOURCE\_*}. Among NNs, Deep MLP barely edges Shallow MLP
  (0.7416 vs 0.7405); both lag classic shallow learners by $\sim$1--2 AUC pts.

  \medskip
  \hp{Imbalance handling on the MLP}
  \begin{itemize}
    \item Threshold tuning lifts recall \hi{0\% $\to$ 41\%}; SMOTE hurts AUC.
    \item Choosing the threshold matters more than rebalancing the data.
  \end{itemize}

  \medskip
  \hp{Limitations}
  \begin{itemize}
    \item Only main \texttt{application\_*.csv} tables  (auxiliary tables would add several AUC points).
    \item Single train/val split (PredefinedSplit), not full $k$-fold CV.
  \end{itemize}
\end{frame}
