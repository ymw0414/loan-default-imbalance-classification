% --------------------------- Slide 2 — Problem ---------------------------
\begin{frame}{Problem \& Motivation}
  \textbf{\color{gwnavy}\large The task}\\
  Predict the probability that a loan applicant defaults
  (binary classification, \texttt{TARGET}~$\in \{0, 1\}$).

  \bigskip
  \textbf{\color{gwnavy}\large Why class imbalance matters}
  \begin{itemize}
    \item Only $\sim$\textbf{8.07\%} of applicants default --- the data is severely imbalanced.
    \item A "predict-no-default" baseline scores 92\% accuracy with \textbf{zero recall}.
    \item Even a model with ROC AUC = 0.745 can classify \emph{zero} defaulters at the 0.5 threshold.
  \end{itemize}

  \bigskip
  \textbf{\color{gwnavy}\large Research question}\\
  {\color{gwbuff}\itshape\large Do classic shallow learners and neural networks
  respond differently to class imbalance handling?}
\end{frame}

% --------------------------- Slide 3 — Dataset ---------------------------
\begin{frame}{Dataset · Home Credit Default Risk}
  \begin{columns}[T,onlytextwidth]
    \column{0.55\textwidth}
      \textbf{\color{gwnavy}Size}
      \begin{itemize}\small
        \item Train: \textbf{307{,}511} applicants $\times$ 122 features
        \item Test: 48{,}744 applicants (Kaggle hidden labels)
      \end{itemize}

      \medskip
      \textbf{\color{gwnavy}Class imbalance}
      \begin{itemize}\small
        \item Default rate $\approx$ \textbf{8.07\%} (24{,}825 / 307{,}511)
      \end{itemize}

      \medskip
      \textbf{\color{gwnavy}Strongest signals}
      \begin{itemize}\small
        \item \texttt{EXT\_SOURCE\_1/2/3} external credit scores ($|\rho| \approx 0.16$--$0.18$)
        \item \texttt{DAYS\_EMPLOYED}\,$=\,$365{,}243 sentinel flags 18\% (unemployed)
        \item Building-info columns are 50--70\% missing
      \end{itemize}

    \column{0.43\textwidth}
      \vspace{0.3em}
      \includegraphics[width=\linewidth]{../result/home_credit/figure/class_balance.png}
  \end{columns}
\end{frame}

% --------------------------- Slide 4 — Methodology ---------------------------
\begin{frame}{Methodology · Shared Pipeline}
  \textbf{\color{gwnavy}\large Preprocessing (identical for all five models)}
  \begin{itemize}
    \item \textbf{80/20 stratified} train--val split (\texttt{random\_state = 42})
    \item Drop \texttt{SK\_ID\_CURR}; keep IDs only for the submission file
    \item Mean imputation for numeric NaN; one-hot encode 16 categoricals $\to$ \textbf{244 features}
    \item \texttt{StandardScaler} fitted on train only (essential for LR / MLP)
  \end{itemize}

  \bigskip
  \textbf{\color{gwnavy}\large Tuning \& validation}
  \begin{itemize}
    \item \texttt{GridSearchCV} with \texttt{PredefinedSplit} (single train$\to$val fold, no random k-fold)
    \item Scoring: \texttt{roc\_auc}; best parameter setting picked per model
    \item CV CSVs saved to \texttt{result/home\_credit/cv\_results/GridSearchCV/}
  \end{itemize}
\end{frame}

% --------------------------- Slide 7 — Neural Network ---------------------------
\begin{frame}{Neural Network Track}
  \textbf{\color{gwnavy}\large Architecture}
  \begin{itemize}\small
    \item \textbf{Shallow MLP} — hidden $(64,)$ \quad $\sim$15.7K parameters
    \item \textbf{Deep MLP} — hidden $(128, 64, 32)$ \quad $\sim$42K parameters
    \item ReLU activation, Adam optimizer, \texttt{early\_stopping = True}
  \end{itemize}

  \bigskip
  \textbf{\color{gwnavy}\large Hyperparameter tuning}
  \begin{itemize}\small
    \item \texttt{alpha} (L2 reg.) $\in \{10^{-4},\,10^{-3}\}$,\quad
          \texttt{lr\_init} $\in \{0.001,\,0.005\}$
    \item \textbf{Best} for both: $\alpha = 10^{-3}$, \texttt{lr} $= 0.005$
  \end{itemize}

  \bigskip
  \textbf{\color{gwnavy}\large Findings}
  \begin{itemize}\small
    \item Shallow $0.7405$ \quad vs \quad Deep $0.7416$ \quad $\Rightarrow$ depth adds $\sim 0.001$ AUC (noise).
    \item NN ranks 4th/5th overall — bottleneck is the \emph{feature set}, not the model.
  \end{itemize}
\end{frame}

% --------------------------- Slide 9 — Conclusions ---------------------------
\begin{frame}{Conclusions}
  \textbf{\color{gwnavy}\large What we learned}
  \begin{itemize}\small
    \item \textbf{HistGradientBoosting} wins at val ROC AUC $= 0.7595$ — tree boosting takes the lead by $\sim$1 AUC point.
    \item \textbf{Neural networks} tie LR and RF — depth alone does not unlock new performance on this tabular data.
    \item {\color{gwbuff}\textbf{Threshold tuning} on the MLP lifts recall from 0\% to 40\%; SMOTE hurt AUC.}
  \end{itemize}

  \bigskip
  \textbf{\color{gwnavy}\large Limitations}
  \begin{itemize}\small
    \item Only main \texttt{application\_*.csv} tables — auxiliary tables typically add several AUC points.
    \item \texttt{PredefinedSplit} is fast but optimistic; full $k$-fold CV would tighten confidence intervals.
  \end{itemize}

  \bigskip
  \textbf{\color{gwnavy}\large Future work}
  \begin{itemize}\small
    \item Integrate auxiliary tables \,·\, ensemble HGBC $+$ best MLP \,·\, Platt / isotonic calibration.
  \end{itemize}
\end{frame}
