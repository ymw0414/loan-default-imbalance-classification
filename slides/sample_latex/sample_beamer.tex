\begin{frame}{Problem \& Motivation \speaker{Minwoo · 1 min}}
  \textbf{The task}\par
  Predict the probability that a loan applicant defaults
  (binary, \texttt{TARGET}~$\in \{0, 1\}$).
  \medskip
  \textbf{Why class imbalance matters}
  \begin{itemize}
    \item Only $\sim$8.07\% of applicants default --- the data is severely imbalanced.
    \item A "predict-no-default" baseline scores 92\% accuracy with zero recall.
    \item A model can hit ROC AUC = 0.745 yet classify zero defaulters at the 0.5 threshold.
  \end{itemize}
  \medskip
  \textbf{Research question}\par
  {\color{accent}\itshape Do classic shallow learners and neural networks
  respond differently to class imbalance handling?}
\end{frame}

% --------------------------- Slide 3 ---------------------------------

  \begin{columns}[T]
    \column{0.55\textwidth}
      \textbf{Size}
      \begin{itemize}\small
        \item Train: 307{,}511 applicants $\times$ 122 features
        \item Test: 48{,}744 applicants (Kaggle hidden labels)
      \end{itemize}
      \textbf{Class imbalance}
      \begin{itemize}\small
        \item Default rate $\approx$ 8.07\% (24{,}825 / 307{,}511)

% --------------------------- Slide 4 ---------------------------------
\begin{frame}{Methodology · Shared Pipeline \speaker{Nathan · 1 min}}
  \textbf{Preprocessing (identical for all five models)}
  \begin{itemize}\small
    \item 80\% / 20\% stratified train--val split (\texttt{random\_state = 42})
    \item Drop \texttt{SK\_ID\_CURR}; keep IDs only for the submission file
    \item Mean imputation for numeric NaN; one-hot encode 16 categoricals $\to$ 244 features
    \item \texttt{StandardScaler} fitted on train only (essential for LR / MLP)
  \end{itemize}
  \medskip
  \textbf{Tuning \& validation}
  \begin{itemize}\small
    \item \texttt{GridSearchCV} with \texttt{PredefinedSplit} (single train$\to$val fold, no random k-fold)
    \item Scoring: \texttt{roc\_auc}; best parameter setting picked per model
    \item CV CSVs written to \texttt{result/home\_credit/cv\_results/GridSearchCV/}
  \end{itemize}
\end{frame}

\begin{frame}{Neural Network Track \speaker{Minwoo · 1.5 min}}
  \textbf{Architecture}
  \begin{itemize}\small
    \item Shallow MLP --- hidden $(64,)$ \quad $\sim$15.7K parameters
    \item Deep MLP --- hidden $(128, 64, 32)$ \quad $\sim$42K parameters
    \item ReLU activation, Adam optimizer, \texttt{early\_stopping = True}
  \end{itemize}
  \textbf{Hyperparameter tuning}
  \begin{itemize}\small
    \item \texttt{alpha} (L2 reg.) $\in \{10^{-4}, 10^{-3}\}$,
          \texttt{lr\_init} $\in \{0.001, 0.005\}$
    \item Best for both: $\alpha = 10^{-3}$, lr = 0.005
  \end{itemize}
  \textbf{Findings}
  \begin{itemize}\small
    \item Shallow $0.7405$ \quad vs \quad Deep $0.7416$ \quad $\Rightarrow$ depth adds $\sim$0.001 (noise).
    \item NN ranks 4th/5th overall; bottleneck is the feature set, not the model.
  \end{itemize}
\end{frame}

% --------------------------- Slide 9 ---------------------------------
\begin{frame}{Conclusions \speaker{Both · 1 min}}
  \textbf{What we learned}
  \begin{itemize}\small
    \item Best: HistGradBoost at val ROC AUC = 0.7595; tree boosting wins by $\sim$1 AUC point.
    \item Neural networks tie LR and RF --- depth alone does not help on this tabular data.
    \item \textcolor{accent}{On the MLP, threshold tuning lifts recall from 0\% to 40\%; SMOTE hurt AUC.}
  \end{itemize}
  \textbf{Limitations}
  \begin{itemize}\small
    \item Only main \texttt{application\_*.csv} tables --- auxiliary tables typically add several AUC points.
    \item \texttt{PredefinedSplit} is fast but optimistic; full $k$-fold CV would tighten confidence intervals.
  \end{itemize}
  \textbf{Future work}
  \begin{itemize}\small
    \item Integrate auxiliary tables; ensemble HGBC + best MLP; Platt / isotonic calibration.
  \end{itemize}
\end{frame}
